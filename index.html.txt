  <script>
    const mensajes = [
      "Dale click si Ender es un idiota",
      "Dale click si no quieres saber nada de él",
      "Dale click si es un tonto",
      "Dale click si con sus tonterías te ha hecho reír",
      "Dale click si te ha ayudado a sentirte cuando todo estaba mal",
      "Dale click si lo detestas, es hombre después de todo",
      "Dale click si sabes que siempre está intentando mejorar para ser un buen hombre para ti",
      "Dale click si… ¿Lo amas?",
      "Ahora dale click, si:",
      "Perdón, sé que soy un tonto, pero este tonto te ama con locura, no quiero perder al amor de mi vida",
      "Quiero que seas tú en todas las vidas, pero como nadie sabe si hay otras, quiero que seas tú el resto que queda de esta, hasta viejitos, como dijimos ❤️"
    ];

    let index = 0;
    const mensaje = document.getElementById("mensaje");
    const opciones = document.getElementById("opciones");
    const final = document.getElementById("final");

    mensaje.addEventListener("click", () => {
      if (index < mensajes.length) {
        mensaje.textContent = mensajes[index];
        index++;
      } else {
        mensaje.style.display = "none";
        opciones.style.display = "block";
      }
    });

    document.querySelectorAll(".btn").forEach(btn => {
      btn.addEventListener("click", () => {
        opciones.style.display = "none";
        final.style.display = "block";
        lanzarCorazones();
      });
    });

    // Corazones de fondo
    for (let i = 0; i < 40; i++) {
      let corazon = document.createElement("div");
      corazon.classList.add("corazon");
      corazon.textContent = "❤";
      corazon.style.left = Math.random() * 100 + "vw";
      corazon.style.top = Math.random() * 100 + "vh";
      corazon.style.fontSize = (Math.random() * 15 + 10) + "px";
      corazon.style.animationDuration = (Math.random() * 4 + 4) + "s";
      document.body.appendChild(corazon);
    }

    // Corazones explosivos al final
    function lanzarCorazones() {
      for (let i = 0; i < 40; i++) {
        let corazon = document.createElement("div");
        corazon.classList.add("explosion");
        corazon.textContent = "❤";
        corazon.style.left = "50%";
        corazon.style.top = "50%";
        corazon.style.setProperty("--x", Math.random());
        corazon.style.setProperty("--y", Math.random());
        document.body.appendChild(corazon);
        setTimeout(() => corazon.remove(), 2000);
      }
    }
  </script>
